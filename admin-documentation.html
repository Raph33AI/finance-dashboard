Système d'authentification

Protection des pages

Toutes les pages sensibles sont protégées par auth-guard.js qui vérifie la présence d'un utilisateur authentifié. En cas d'absence d'authentification, redirection automatique vers la page de login.

JavaScript - auth-guard.js 
// auth-guard.js
firebase.auth().onAuthStateChanged(function(user) {
    if (!user) {
        // Utilisateur non connecté
        const publicPages = ['index.html', 'landing.html', 'login.html', 'register.html'];
        const currentPage = window.location.pathname.split('/').pop();
        
        if (!publicPages.includes(currentPage)) {
            // Redirection vers login
            window.location.href = 'login.html';
        }
    }
});

Ensemble Prediction System

Système de prédiction combinant 3 approches différentes pour améliorer la précision :

JavaScript - Ensemble Models 
// 1. LINEAR REGRESSION
function linearRegressionPrediction(prices) {
    const n = prices.length;
    let sumX = 0, sumY = 0, sumXY = 0, sumX2 = 0;
    
    for (let i = 0; i < n; i++) {
        sumX += i;
        sumY += prices[i];
        sumXY += i * prices[i];
        sumX2 += i * i;
    }
    
    const slope = (n * sumXY - sumX * sumY) / (n * sumX2 - sumX * sumX);
    const intercept = (sumY - slope * sumX) / n;
    
    return intercept + slope * n; // Prédiction pour le prochain point
}

// 2. MOVING AVERAGE CROSSOVER
function movingAveragePrediction(prices) {
    const ma50 = calculateMA(prices, 50);
    const ma200 = calculateMA(prices, 200);
    
    const currentPrice = prices[prices.length - 1];
    const ma50Trend = ma50[ma50.length - 1] - ma50[ma50.length - 5];
    const ma200Trend = ma200[ma200.length - 1] - ma200[ma200.length - 5];
    
    if (ma50Trend > 0 && ma200Trend > 0) {
        return currentPrice * 1.05; // Bullish
    } else if (ma50Trend < 0 && ma200Trend < 0) {
        return currentPrice * 0.95; // Bearish
    } else {
        return currentPrice; // Neutral
    }
}

// 3. RSI-BASED PREDICTION
function rsiBasedPrediction(prices) {
    const rsi = calculateRSI(prices, 14);
    const currentRSI = rsi[rsi.length - 1];
    const currentPrice = prices[prices.length - 1];
    
    if (currentRSI < 30) {
        return currentPrice * 1.08; // Oversold → Bullish
    } else if (currentRSI > 70) {
        return currentPrice * 0.92; // Overbought → Bearish
    } else {
        return currentPrice * 1.02; // Neutral → Slight bullish
    }
}

// ENSEMBLE FINAL (Moyenne pondérée)
function ensemblePrediction(prices) {
    const pred1 = linearRegressionPrediction(prices);
    const pred2 = movingAveragePrediction(prices);
    const pred3 = rsiBasedPrediction(prices);
    
    // Poids : Linear 40%, MA 35%, RSI 25%
    return (pred1 * 0.4) + (pred2 * 0.35) + (pred3 * 0.25);
}

Monte Carlo Simulation

JavaScript - Monte Carlo 
function monteCarloSimulation(initialValue, expectedReturn, volatility, days, simulations) {
    const results = [];
    
    for (let i = 0; i < simulations; i++) {
        let value = initialValue;
        const path = [value];
        
        for (let day = 0; day < days; day++) {
            // Générer rendement aléatoire (distribution normale)
            const randomReturn = boxMullerTransform(expectedReturn / 252, volatility / Math.sqrt(252));
            value *= (1 + randomReturn);
            path.push(value);
        }
        
        results.push({
            finalValue: value,
            path: path,
            return: ((value - initialValue) / initialValue) * 100
        });
    }
    
    // Calculs statistiques
    const finalValues = results.map(r => r.finalValue).sort((a, b) => a - b);
    
    return {
        simulations: results,
        statistics: {
            mean: average(finalValues),
            median: finalValues[Math.floor(finalValues.length / 2)],
            percentile5: finalValues[Math.floor(finalValues.length * 0.05)],
            percentile25: finalValues[Math.floor(finalValues.length * 0.25)],
            percentile75: finalValues[Math.floor(finalValues.length * 0.75)],
            percentile95: finalValues[Math.floor(finalValues.length * 0.95)],
            stdDev: standardDeviation(finalValues),
            var95: initialValue - finalValues[Math.floor(finalValues.length * 0.05)]
        }
    };
}

// Box-Muller Transform pour distribution normale
function boxMullerTransform(mean, stdDev) {
    const u1 = Math.random();
    const u2 = Math.random();
    const z0 = Math.sqrt(-2 * Math.log(u1)) * Math.cos(2 * Math.PI * u2);
    return mean + stdDev * z0;
}

Risk Parity Optimization

JavaScript - Risk Parity 
/**
 * Optimisation Risk Parity
 * Objectif : Équilibrer la contribution au risque de chaque actif
 */
function riskParityOptimization(returns, targetRisk = 0.10) {
    const n = returns[0].length; // Nombre d'actifs
    
    // 1. Calcul de la matrice de covariance
    const covMatrix = calculateCovarianceMatrix(returns);
    
    // 2. Initialisation des poids (equal-weighted)
    let weights = new Array(n).fill(1 / n);
    
    // 3. Optimisation itérative (Newton-Raphson)
    const maxIterations = 1000;
    const tolerance = 1e-6;
    
    for (let iter = 0; iter < maxIterations; iter++) {
        // Calcul du risque de portefeuille
        const portfolioRisk = Math.sqrt(
            weights.reduce((sum, w, i) => 
                sum + weights.reduce((s, w2, j) => 
                    s + w * w2 * covMatrix[i][j], 0), 0)
        );
        
        // Calcul des contributions au risque
        const riskContributions = weights.map((w, i) => 
            (w * weights.reduce((s, w2, j) => s + w2 * covMatrix[i][j], 0)) / portfolioRisk
        );
        
        // Target contribution (égale pour tous)
        const targetContribution = portfolioRisk / n;
        
        // Ajustement des poids
        const adjustmentFactor = riskContributions.map(rc => 
            targetContribution / rc
        );
        
        weights = weights.map((w, i) => w * adjustmentFactor[i]);
        
        // Normalisation (somme = 1)
        const sum = weights.reduce((a, b) => a + b, 0);
        weights = weights.map(w => w / sum);
        
        // Test de convergence
        const maxDiff = Math.max(...riskContributions.map(rc => 
            Math.abs(rc - targetContribution)
        ));
        
        if (maxDiff < tolerance) break;
    }
    
    return {
        weights: weights,
        expectedReturn: calculatePortfolioReturn(returns, weights),
        volatility: calculatePortfolioVolatility(covMatrix, weights),
        sharpeRatio: calculateSharpeRatio(returns, weights)
    };
}
